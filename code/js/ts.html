# 安装ts

npm install -g typescript
1

安装完成后，在控制台运行如下命令，检查安装是否成功(3.x)：

tsc -V
1

# 编译ts文件为js

tsc ./xxx.ts
1

TIP

1如果ts中直接书写js语法的代码，那么在html文件中直接就可以引入ts文件，在谷歌浏览器中是可以直接使用的
2如果ts中有了ts的语法代码，那么需要把这个ts文件编译成js文件，在html文件中引入js的文件来使用
3如果ts文件中的函数中的形参，如果使用了某个类型进行修饰，那么最终再编译后的js文件中是没有这个类型的
4ts中的变量使用let进行修饰，编译后的js文件中的修饰符就变成var

# ts在vscode中自动编译

1 生成配置文件tsconfig.json

tsc --init
1

2 把tsconfig.json文件中"outDir": "./", 改成"outDir": "./js", 为js输出目录

"strict": true改成"strict": false, 不使用严格模式

3 启动监视任务: 终端 -> 运行任务 -> 监视tsconfig.json

# 接口

接口：一种能力，一种约束

(()=>{
    //定义一个接口
    interface IPerson{
        firstName:string,//姓
        lastName:string//名
    }
    //输出姓名
    function showFullName(person:IPerson){
        return person.firstName+'_'+person.lastName
    }
    const person={
        firstName:'李',
        lastName:'某'
    }
    
    console.log(showFullName(person));
    
})()

# 类

ts中书写js中的类，演示效果

(()=>{
    //定义一个接口
    interface IPerson{
        firstName:string,//姓
        lastName:string//名
    }
    //定义一个类
    class Person{
        //定义公共的字段(属性)
        firstName:string//姓
        lastName:string//名
        fullName:string//姓名
        //定义一个构造器函数
        constructor(firstName:string,lastName:string){
            //更新属性数据
            this.firstName = firstName
            this.lastName = lastName
            //姓名
            this.fullName = this.firstName+'_'+this.lastName
        }
    }

    //定义个函数
    function showFullName(person:IPerson){
        return person.firstName+'-'+person.lastName
    }
    //实例化对象
    const person = new Person('李','某')
    console.log(showFullName(person));

})()
