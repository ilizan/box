字符串替换正则

var str = "1111111"  
str.replace(/1/, '2')//"2111111"
str.replace(/1/gi, '2')//"2222222"

var html = "<img src='#'>"
html.replace(/\<img/gi, '<img style="width:100%;height:auto"')//<img style=\"width:100%;height:auto\" src='#'>

var html2 = "<div><div>111232</div></div>"
function delHtmlTag(str)
{
      return str.replace(/<[^>]+>/g,"");//去掉所有的html标记
}
delHtmlTag(html2)//111232

正则表达式

正则表达式中/i,/g,/ig,/gi,/m的区别和含义
/i (忽略大小写)
/g (全文查找出现的所有匹配字符)
/m (多行查找)
/gi(全文查找、忽略大小写)
/ig(全文查找、忽略大小写)

时间戳转年月日格式

getMyData:function(posttime){
    var time = posttime*1000
    var d = new Date(time)
    var year = d.getFullYear()
    var mon = (d.getMonth()+1)<10?'0'+(d.getMonth()+1): d.getMonth()+1
    var day = (d.getDate()+1)<10?'0'+(d.getDate()+1): d.getDate()+1
    return year+'-'+mon+'-'+day
}

js排序

let arr = [
    {
        id:1,
        click:20
    },
    {
        id:3,
        click:30
    },
    {
        id:2,
        click:25
    }
]
// order(field,type='asc')
function order(field,type='asc'){
    return function(a,b){
        if(type == 'asc'){
            return a[field]>b[field]?1:-1
        }else{
            return a[field]>b[field]?-1:1
        }
    }
}
let hd = arr.sort(order('id','desc'))//排序的字段名
