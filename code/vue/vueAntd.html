# 多层级vue Router递归取出最底层

function flattenDeep(routeArr, arr=[]){
  routeArr.forEach(route => {
      if (route.children && Array.isArray(route.children)) {
          flattenDeep(route.children, arr)
      } else {
          arr.push(route)
      }
  })
  return arr
}

function addRouterFn(PName,menuList){
  var list = flattenDeep(menuList)
  list.forEach(item => {
    //文件名转换为大写字母
    let fileUrl = 
      item.path.substr(0,item.path.lastIndexOf('/'))+'/' +
      item.path.substr(item.path.lastIndexOf('/')+1,1).toLocaleUpperCase() +
      item.path.substr(item.path.lastIndexOf('/')+2,item.path.length)
    router.addRoute(PName.substr(0,1).toLocaleUpperCase()+PName.substr(1,PName.length), {
      path: item.path, // 这里要把父路由的路径也带上
      name: item.name,
      component: () => import('../views/page'+fileUrl+'.vue')
      
    })

  });
  router.addRoute(PName.substr(0,1).toLocaleUpperCase()+PName.substr(1,PName.length), {
    path: '/'+PName+'/*', // 这里要把父路由的路径也带上
    name: '404',
    component: () => import('../views/page/404/404')
  })
}

# 创建标签页tags

思路：watch监听当前路由变化（标签tags组件内），把当前路由存到vuex中为数组，创建标签页组件，循环该存标签页的数组；选中当前标签页后，打开选中标签的路由

vuex中：

//标签页
mainTabs:[{"path":"/main/home","params":{},"query":{},"name":"Home","title":"首页",closable: false}],//首页不能关闭
mainTabsActiveName: ''//当前标签页name，选中tabs使用

//标签页
updateMainTabs(state, tabs) {
    state.mainTabs = tabs
},
updateMainTabsActiveName(state, name) {
    state.mainTabsActiveName = name
}

新建tabs组件：

<template>
  <div class="mainTabs">
    <div class="tabsAll">
      <a-tabs v-model="mainTabsActiveName" type="editable-card" @tabClick="selectTab" @edit="onEdit" :hideAdd="true">
        <a-tab-pane v-for="(item) in mainTabs" :key="item.path" :tab="item.title" :closable="item.closable"/>
        
      </a-tabs>
    </div>
  </div>
</template>

<script>
export default {
  name: "mainTabs",
  computed: {
    mainTabs: {
      get(){
        return this.$store.state.common.mainTabs
      },
      set(val){
        this.$store.commit('common/updateMainTabs',val)
      }
    },
    mainTabsActiveName: {
      get() {
        return this.$store.state.common.mainTabsActiveName
      },
      set(val) {
        this.$store.commit('common/updateMainTabsActiveName', val)
      }
    }
  },
  watch: {
    '$route'(newValue, oldValue) {
      this.routeHandle(newValue);
    }
  },
  created () {
    this.routeHandle(this.$route);
  },
  methods: {
    // tabs, 删除tab
    removeTabHandle(targetKey) {
      this.mainTabs = this.mainTabs.filter(item=>item.path != targetKey)
      if(this.mainTabs.length >= 1){
        //当前选中tab被删除
        if(targetKey == this.mainTabsActiveName){
          var tab = this.mainTabs[this.mainTabs.length - 1]
          this.$router.push(
            { path:tab.path, query: tab.query, params: tab.params },
            () => {
              this.mainTabsActiveName = this.$route.path
            }
          )
        }
      }else{

      }
    },
    //选中当前
    selectTab(val){
      this.$router.push(val)
    },
    //关闭当前
    onEdit(targetKey, action) {
      this.removeTabHandle(targetKey)
    },
    //路由操作
    routeHandle(route) {
      var tab = this.mainTabs.filter(item=>item.name == route.name)[0]
      //tabs中无当前页面路由信息，添加
      if(!tab){
        tab = {
          path: route.path,
          params: route.params,
          query: route.query,
          name: route.name,
          title:route.meta.title
        }
        this.mainTabs = this.mainTabs.concat(tab)
      }
      //添加后或有当前路由，选中当前路由，给当前路由赋值
      this.mainTabsActiveName = route.path
      
      
    }
  },
};
</script>
