<template>
    <el-dropdown
      ref="dropdown"
      trigger="click"
      @command="handleCommand"
      placement="bottom-start"
      @visible-change="handleVisibleChangeEvent"
    >
      <span class="el-dropdown-link">
        <span class="tree-title">{{ selectedLabel }}</span>
        <i class="el-icon-arrow-down el-icon--right icon-right"></i>
      </span>
      <el-dropdown-menu slot="dropdown">
        <div class="input-box">
            <el-input
                size="small"
                placeholder="输入关键字进行过滤"
                v-model="filterText">
            </el-input>
        </div>
        <div class="tree-cu-box">
            <el-tree
                :data="treeData"
                node-key="id"
                :expand-on-click-node="false"
                :check-on-click-node="true"
                :show-checkbox="false"
                highlight-current
                :props="defaultProps"
                @node-click="handleNodeClick"
                ref="tree"
                :filter-node-method="filterNode"
                :default-expanded-keys="defaultExpandedKeys">
                <template #default="{ node, data }">
                    <el-radio
                        :value="value"
                        :label="data.id"
                        @change="handleRadioChange(data)"
                    >
                        {{ data.title }}
                    </el-radio>
                </template>
            </el-tree>
        </div>
      </el-dropdown-menu>
    </el-dropdown>
  </template>
  
  <script>
  export default {
    name: 'DropdownTree',
    props: {
      value: {
        type: [String, Number],
        default: ''
      },
      data: {
        type: Array,
        required: true
      }
    },
    data() {
      return {
        filterText: '',
        selectedLabel: '',
        treeData: [],
        defaultProps: {
          children: 'arrchildid',
          label: 'title'
        },
        defaultExpandedKeys:[]//默认展开
      };
    },
    watch: {
        data: {
            immediate: true,
            deep: true,
            handler(newValue, oldValue) {
                let data = this.normalizeTree(newValue)
                this.treeData = data
            }
        },
        value(val) {
            this.setSelectedLabel(val);
        },
        filterText(val) {
            this.$refs.tree.filter(val);
        }
    },
    methods: {
        handleVisibleChangeEvent(visible){
            this.defaultExpandedKeys = []
            if (visible) {
                this.defaultExpandedKeys = [this.value]
            } else {
                console.log('隐藏');
            }
            
        },
        filterNode(value, data) {
            if (!value) return true;
            return data.title.indexOf(value) !== -1;
        },
        normalizeTree(tree) {
            // 递归函数，用于遍历树形结构  
            function traverse(node) {  
                // 检查arrchildid是否是字符串  
                if (typeof node.arrchildid === 'string') {  
                    // 如果是字符串，则替换为空数组  
                    node.arrchildid = [];  
                } else if (Array.isArray(node.arrchildid)) {  
                    // 如果arrchildid是数组，则递归处理子节点  
                    node.arrchildid.forEach(traverse);  
                }  
            }
            // 遍历树的每个节点  
            tree.forEach(traverse);
            // 返回修改后的树  
            return tree;  
        },
      handleCommand(command) {
        this.$emit('input', command);
      },
      handleRadioChange(data) {
        console.log(data);
        this.selectedLabel = data.title;
        this.$emit('input', data.id);
        this.$nextTick(() => {
            this.$refs.dropdown.hide();
        });
      },
      handleNodeClick(data) {
        this.selectedLabel = data.title;
        this.$emit('input', data.id);
        this.$nextTick(() => {
            this.$refs.dropdown.hide();
        });
      },
      setSelectedLabel(value) {
        console.log(value);
        const findLabel = (nodes, value) => {
          for (let node of nodes) {
            if (node.id == value) {
              return node.title;
            }
            if (node.arrchildid) {
              const result = findLabel(node.arrchildid, value);
              if (result) {
                return result;
              }
            }
          }
          return '';
        };
        this.$refs.tree.setCheckedKeys([value])
        setTimeout(() => {
            this.selectedLabel = findLabel(this.treeData, value) || '';
        }, 500);
      },
    },
    mounted() {
      this.setSelectedLabel(this.value);
    }
  };
  </script>
