vant组件按需引入

按照vant官网安装babel-plugin-import组件:

npm i babel-plugin-import -D

babel.config.js中配置:

plugins: [
    ['import', {
      libraryName: 'vant',
      libraryDirectory: 'es',
      style: true,
      // 指定样式路径
      //style: (name) => `${name}/style/less`,//这步在下面vant定制主题会用到
    }, 'vant']
  ]

再在main.js中配置：

import { Button } from 'vant';Vue.use(Button);

或者在src目录下新建plugins文件夹，新建vant.js文件：

import Vue from 'vue'
import { Button } from 'vant';
Vue.use(Button);

再在main.js中引入vant.js：

//按需引入vant
import './plugins/vant'

rem适配

postcss-pxtorem 是一款 postcss 插件，用于将单位转化为 rem

lib-flexible 用于设置 rem 基准值

npm i postcss-pxtorem lib-flexible -D

当前版本号

"postcss-pxtorem": "^5.1.1", "lib-flexible": "^0.3.2",

然后在vue.config.js里配置:https://www.cnblogs.com/yifeng555/p/12734032.html

const css = {
  loaderOptions: {
    postcss: {
      plugins: [
        require("autoprefixer")({
          // 配置使用 autoprefixer
          // overrideBrowserslist: ['Android >= 4.0', 'iOS >= 8', 'Chrome > 31']
          overrideBrowserslist: ['last 2 version', '>1%', 'ios 7']
        }),
        require("postcss-pxtorem")({
          rootValue: 37.5,
          propList: ['*']
        })
      ]
    }
  }
}
module.exports = {
	css
}

DANGER

此时启动项目控制台报错“ Error: PostCSS plugin postcss-pxtorem requires PostCSS 8.” 需要把package.json中"postcss-pxtorem": "^6.0.0"版本号修改为5.1.1， 重新npm i安装依赖，启动项目正常。

然后在main.js里引入:

//rem适配
import "lib-flexible/flexible"

Vant定制主题和less全局样式使用

Vant定制主题

在开发中经常遇到在组件中使用公共文件样式的情况，这时候就需要用到全局配置，而非在每一个组件都单独引用，太繁琐，不易维护

定义一个变量，却需要每个 .vue 文件都需要单独引入这个全局变量才可以，导致很多重复不必要的工作，因而得寻找一种可以任何地方都可以访问的方法。 在 babel.config.js 中配置按需引入样式源文件

//style: true,//修改为
// 指定样式路径
style: (name) => `${name}/style/less`,

const path = require("path");

格式化时间

安装moment：//当前版本号2.29.1

cnpm i moment --save

步骤一(过滤器)

新建moment.js文件

import Vue from 'vue'
// 格式化时间
import moment from 'moment'
Vue.filter('datefmt', function(input, fmtstring) {
  // 使用momentjs这个日期格式化类库实现日期的格式化功能
  if (input) {
    return moment(input).format(fmtstring)
  } else {
    //不存在，数据格式化
    return '--'
  }
})
Vue.filter('datefmtOther', function(input, fmtstring) {
  // 使用momentjs这个日期格式化类库实现日期的格式化功能
  if (input) {
    return moment(input, 'YYYY-MM-DD').format(fmtstring)
  } else {
    //不存在，数据格式化
    return '--'
  }
})

main.js中引入此js

//引入日期时间格式化
import './plugins/moment'

使用方法：

{{new Date() | datefmt('YYYY-MM-DD HH:mm:ss')}}
{{new Date() | datefmtOther('YYYY.MM.DD')}}

步骤二、js中使用：

moment.js中加入

import moment from 'moment'
Vue.prototype.$moment = moment;//赋值使用

组件中使用：

this.$moment(new Date()).format("YYYY-MM-DD HH:mm:ss")

添加公共js方法

src/assets/js中新建js文件 util.js

export const isIos = () => {
    var u = navigator.userAgent;
    if (u.indexOf('Android') > -1 || u.indexOf('Linux') > -1) {  //安卓手机
        return false
    } else if (u.indexOf('iPhone') > -1) {//苹果手机
        return true
    } else if (u.indexOf('iPad') > -1) {//iPad
        return false
    } else if (u.indexOf('Windows Phone') > -1) {//winphone手机
        return false
    } else {
        return false
    }
}

export default {
    isIos
}

Main.js中

//js公共方法
import util from './assets/js/util'
Vue.prototype.$util = util

使用：

this.$util.isIos()

路由钩子做登录判断

router.js中添加：

router.beforeEach((to, from, next) => {
  if (to.meta.requireAuth) {
    // 判断该路由是否需要登录权限
    if (localStorage.getItem('isLogin')) {
      next()
    } else {
        next({
          path: '/login'//登录页
        })
    }
  } else {
    next()
  }
})

修改路由：

{
    path: "/",
    name: "Home",
    component: Home,
    meta: {
      requireAuth: true // 添加该字段，表示进入这个路由是需要登录的
    }
  },
  {
    path: "/login",
    name: "login",
    component: Login
  },

login登陆成功后 保存localStorage.setItem('isLogin',true)再跳转到首页即可

localStorage.setItem('isLogin',true);
this.$router.push('/')

首页添加tabbar

首先引入vant的Tabbar, TabbarItem组件 App.vue中添加：

<van-tabbar v-if="$route.meta.showTab" v-model="active" route id="mtTabbar">
  <van-tabbar-item to="/" icon="wap-home">主页</van-tabbar-item>
  <van-tabbar-item to="/about" icon="manager">我</van-tabbar-item>
</van-tabbar>

部分说明:

v-model="active",data里配置active:'home'

route:开启路由模式

to='/xxx' 对应路由js中配置

V-if: router.js中配置showTab:true

在router.js中 显示底部菜单的页面添加

{
  path: "/home",
  name: "Home",
  component: Home,
  meta: {
    requireAuth: true, // 添加该字段，表示进入这个路由是需要登录的

  }
