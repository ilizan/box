# 展开运算符

arr1 = [1,2,3]
arr2 = ['a','b','c']
arr3 = arr1.concat(arr2)
arr4 = [...arr1,...arr2]

console.log(arr3) //[1, 2, 3, "a", "b", "c"]
console.log(arr4) //[1, 2, 3, "a", "b", "c"]
